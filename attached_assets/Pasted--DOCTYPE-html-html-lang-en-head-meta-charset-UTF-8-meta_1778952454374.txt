let currentFilter = 'all';
let deck = [];
let index = 0;
let done = 0;
let again = 0;
let revealed = false;
let againCards = [];

function buildDeck() {
  if(currentFilter === 'all') deck = [...cards];
  else if(currentFilter === 'again') deck = [...againCards];
  else deck = cards.filter(c => c.cat === currentFilter);
  index = 0; done = 0; again = 0; revealed = false;
  render();
}

function setFilter(f) {
  currentFilter = f;
  document.querySelectorAll('.fb').forEach(b => b.classList.remove('on'));
  const btns = document.querySelectorAll('.fb');
  btns.forEach(b => { if(b.dataset.cat === f) b.classList.add('on'); });
  buildDeck();
}

function render() {
  const main = document.getElementById('mainArea');
  document.getElementById('totNum').textContent = deck.length;
  document.getElementById('doneNum').textContent = done;
  document.getElementById('againNum').textContent = again;

  if(deck.length === 0) {
    main.innerHTML = `<div class="done-screen"><h2>No cards in this set.</h2><p>Try a different filter.</p></div>`;
    document.getElementById('pbar').style.width = '0%';
    return;
  }

  if(index >= deck.length) {
    const pct = Math.round((done / deck.length) * 100);
    main.innerHTML = `
      <div class="done-screen">
        <h2>Set complete 🎉</h2>
        <p>You've been through all ${deck.length} cards in this set.</p>
        <div class="done-stats">
          <div class="done-stat"><span class="num num-green">${done}</span><span class="lbl">Got it</span></div>
          <div class="done-stat"><span class="num num-red">${again}</span><span class="lbl">Again</span></div>
          <div class="done-stat"><span class="num" style="color:var(--accent)">${pct}%</span><span class="lbl">Score</span></div>
        </div>
        ${againCards.length > 0 ? `<button class="btn btn-reveal" onclick="setFilter('again')">Repeat missed cards (${againCards.length})</button>` : ''}
        <button class="btn" style="margin-left:8px" onclick="setFilter(currentFilter)">Restart</button>
      </div>`;
    document.getElementById('pbar').style.width = '100%';
    document.getElementById('curNum').textContent = deck.length;
    return;
  }

  const card = deck[index];
  document.getElementById('curNum').textContent = index + 1;
  document.getElementById('pbar').style.width = ((index / deck.length) * 100) + '%';
  document.getElementById('navInfo').textContent = catLabels[card.cat] || card.cat;

  const aHTML = card.a.map(p => `<li>${p}</li>`).join('');
  main.innerHTML = `
    <div class="card">
      <span class="card-cat cat-${card.cat}">${catLabels[card.cat]||card.cat}</span>
      <div class="question">${card.q}</div>
      <div class="answer-section" id="answerSection">
        <div class="answer-divider"></div>
        <div class="answer-label">Answer</div>
        <ul class="answer-points">${aHTML}</ul>
      </div>
      <div class="btn-row" id="btnRow">
        <button class="btn btn-reveal" onclick="reveal()">Show answer</button>
        <button class="btn btn-skip" onclick="nextCard()">Skip →</button>
      </div>
    </div>`;
  revealed = false;
}

function reveal() {
  document.getElementById('answerSection').classList.add('visible');
  document.getElementById('btnRow').innerHTML = `
    <button class="btn btn-good" onclick="markGood()">✓ Got it</button>
    <button class="btn btn-again" onclick="markAgain()">⟳ Again</button>`;
  revealed = true;
}

function markGood() {
  done++;
  index++;
  render();
}

function markAgain() {
  again++;
  const card = deck[index];
  if(!againCards.find(c => c.q === card.q)) againCards.push(card);
  index++;
  render();
}

function nextCard() {
  index++;
  render();
}

function prevCard() {
  if(index > 0) { index--; render(); }
}
